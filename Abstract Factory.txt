// Abstract Factory
class ModernFurnitureFactory {
  createChair() { return new ModernChair(); }
  createTable() { return new ModernTable(); }
}
class ClassicFurnitureFactory {
  createChair() { return new ClassicChair(); }
  createTable() { return new ClassicTable(); }
}

// Products
class ModernChair { sit() { console.log("Sitting on a modern chair"); } }
class ModernTable { use() { console.log("Using a modern table"); } }
class ClassicChair { sit() { console.log("Sitting on a classic chair"); } }
class ClassicTable { use() { console.log("Using a classic table"); } }

// Client code
function client(factory) {
  const chair = factory.createChair();
  const table = factory.createTable();
  chair.sit();
  table.use();
}

client(new ModernFurnitureFactory());
client(new ClassicFurnitureFactory());
