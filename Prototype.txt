// Prototype
class Enemy {
  constructor(type, health) {
    this.type = type;
    this.health = health;
  }
  clone() {
    return new Enemy(this.type, this.health);
  }
}

const baseEnemy = new Enemy("Orc", 100);
const enemy1 = baseEnemy.clone();
const enemy2 = baseEnemy.clone();

enemy1.health = 80;
console.log(enemy1, enemy2);
