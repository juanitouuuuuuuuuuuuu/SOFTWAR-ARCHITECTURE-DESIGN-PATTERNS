// Visitor
class Circle {
  accept(visitor) { visitor.visitCircle(this); }
}
class Rectangle {
  accept(visitor) { visitor.visitRectangle(this); }
}

class ShapeExporter {
  visitCircle(circle) { console.log("Exporting Circle to SVG"); }
  visitRectangle(rect) { console.log("Exporting Rectangle to SVG"); }
}

const shapes = [new Circle(), new Rectangle()];
const exporter = new ShapeExporter();

shapes.forEach(shape => shape.accept(exporter));
